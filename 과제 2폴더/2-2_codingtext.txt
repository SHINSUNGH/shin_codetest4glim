#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int tmp = 0;
    int row_size = board.size();
    int len_size = board[0].size();
    
    for (int i = 0; i < row_size; ++i) {
        for (int j = 0; j < len_size; ++j) {
            if ( i + j <= k ){
                answer += board[i][j];
            }
        }
    }
    return answer;
}
